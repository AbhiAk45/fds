bal=0
while(1):
    t=input("Enter Transaction \nE-Exit\nD-Diposit Amt\nW-Withdraw Amt:\nTransaction:")
    d=t
    #print(d)
    if(d[0]=='E'):
       break
    if(d[0]=='D'): #deposit
        bal=bal-int(d[1])
    else:
        if(d[0]=='W')and(bal>int(d[1])):
            bal=bal-int(d[1])
        else:
            print("Incorrectt input/ insufficient balance:")
    print("Current Balance:",bal)
